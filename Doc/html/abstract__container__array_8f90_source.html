115  subroutine resizearray(this, newsize, initvalue)
116  class(container_array), intent(inout) :: this
117  integer, intent(in) :: newsize
118  class(*), target, optional, intent(in) :: initvalue
119  class(*), pointer :: value
120  integer :: i, tocopy
121  type(container), allocatable, dimension(:) :: temp
122  value => mc
123  tocopy=min(this%getLength(),newsize)
124  allocate(temp(newsize))
125  do i=1, tocopy
126  call temp(i)%storeContent(this%get(i))
127  enddo
128  if (present(initvalue)) then
129  value => initvalue !pointing to the initial value to store in every entry
130  end if
131  do i= tocopy+1, newsize
132  call temp(i)%storeContent(value)
133  end do
134  call this%init(newsize,source=temp)
135  end subroutine resizearray
